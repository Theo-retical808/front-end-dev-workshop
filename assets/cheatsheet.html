📋 Frontend Development Cheat Sheet

📄 HTML — Structure

Page Template

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <title>My Page</title>
</head>
<body>
    <h1>Hello!</h1>
</body>
</html>

TagPurpose

🎨 CSS — Styling

Selectors

/* Element */
h1 { color: blue; }

/* Class (.) */
.card { padding: 10px; }

/* ID (#) */
#header { background: red; }

/* Multiple */
h1, h2 { font-weight: bold; }

Common Properties

PropertyWhat it does
colorText color
background-colorBackground
font-sizeText size
marginSpace outside
paddingSpace inside
borderBorder line
border-radiusRounded corners
display: flexFlexbox layout
width/heightSize

Flexbox Quick Guide

/* Parent */
.container {
    display: flex;
    justify-content: center;
    align-items: center;
    gap: 10px;
    flex-wrap: wrap;
}

/* Child */
.item {
    flex: 1;
}

⚡ JavaScript — Interaction

Variables & Types

// Variables
let name = "Alice";    // can change
const age = 16;        // can't change

// Types
"hello"    // String (text)
42         // Number
true/false // Boolean
[1, 2, 3]  // Array (list)
{a: 1}     // Object

DOM Interaction

// Get elements
const btn = document.getElementById("my-btn");
const items = document.querySelectorAll(".item");

// Change content
element.textContent = "New text";
element.innerHTML = "<b>Bold</b>";

// Change style
element.style.color = "red";

// Add/remove classes
element.classList.add("active");
element.classList.toggle("hidden");

Events

// Click event
button.addEventListener("click", function() {
    alert("Clicked!");
});

// Input event
input.addEventListener("input", function() {
    console.log(input.value);
});

// Common events:
// "click", "input", "submit",
// "mouseover", "keydown"

Fetch API

// Basic fetch
async function getData() {
    try {
        const response = await fetch(url);
        const data = await response.json();
        console.log(data);
    } catch (error) {
        console.error("Error:", error);
    }
}

💡 Quick Tips

Press F12 to open Developer Tools in your browser. The Console tab is your best friend!
Use console.log("hello") to debug — it prints messages in the Console.
CSS colors: use names (red), hex (#ff0000), or rgb (rgb(255,0,0))
Ctrl+S saves your file. Ctrl+Z undoes changes. Ctrl+Shift+Z redoes.
